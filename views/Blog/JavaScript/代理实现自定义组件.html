代理实现自定义验证组件

代理实现自定义验证组件

summer 2020-04-29 JavaScript
<!DOCTYPE html>
<html lang="en">

<head>
	<meta charset="UTF-8">
	<meta name="viewport" content="width=device-width, initial-scale=1.0">
	<meta http-equiv="X-UA-Compatible" content="ie=edge">
	<title>Document</title>
</head>
<style>
	* {
		padding: 0;
		margin: 0;
	}

	input {
		border: none;
		outline: none;
	}

	body,
	html {
		width: 100%;
		height: 100%;
		display: flex;
		justify-content: center;
		align-items: center;
	}

	.validate {
		width: 300px;
		text-align: center;
	}

	.validate input {
		height: 30px;
		font-size: 20px;
		margin-top: 10px;
		text-indent: 10px;
		border-radius: 5px;
		box-shadow: 0 0 3px rgba(0, 0, 0, .2);
	}

	.err {
		box-shadow: 0 0 3px #eb4d4b !important;
	}

	.ok {
		box-shadow: 0 0 3px #6ab04c !important;
	}
</style>

<body>
	<div class="validate">
		<input type="text" validate rule="max:10,min:6">
		<input type="text" validate rule="max:3,isNumber">
	</div>
	<script>
		"use strict"
		class Validate {
			max(value, len) {
				return value.length <= len;
			}
			min(value, len) {
				return value.length >= len;
			}
			isNumber(value) {
				return /^\d+$/.test(value);
			}
		}
		// 代理工厂
		function ProxyFactory(target) {
			return new Proxy(target, {
				set(target, property, item) {
					const rule = item.getAttribute("rule");
					const validate = new Validate();
					let state = rule.split(",").every(rule => {
						const info = rule.split(":")
						return validate[info[0]](item.value, info[1])
					});
					state ? item.className = "ok" : item.className = "err";
					return true;
				},
				get(target, key) {
					return target[key];
				}
			})
		}
		const proxy = ProxyFactory(
			document.querySelectorAll("[validate]")
		)
		proxy.forEach((item, index) => {
			item.addEventListener("keyup", function () {
				proxy[index] = this;
			})
		});
	</script>
</body>

</html>
